USE sample_Db;


CREATE TABLE EmployeeData (
    ID INT  PRIMARY KEY,
    Name VARCHAR(100),
    Age INT ,
    Salary DECIMAL(10, 2)
);

alter PROCEDURE InsertEmployeeData
    @Name VARCHAR(100),
    @Age INT = 18,
    @Sarlay DECIMAL(10, 2) = 10000.00
AS
BEGIN
    
    IF (@Age >= 18 AND @Age <= 80) AND (@Sarlay >= 10000.00 AND @Sarlay <= 25000.00)
    BEGIN
        INSERT INTO EmployeeData (Name, Age, Salary)
        VALUES (@Name, @Age, @Sarlay);
          PRINT 'Employee data inserted successfully.';
    END
   ELSE
    BEGIN
        PRINT 'Invalid data. Age should be between 18 and 80, and salary should be between 10000.00 and 25000.00.';
    END
END;

CREATE PROCEDURE GetEmployeeDetails
    @EmployeeID INT
AS
BEGIN
    SELECT ID, Name, Age, Salary
    FROM EmployeeData
    WHERE ID = @EmployeeID;
END;

CREATE PROCEDURE UniqueSalary
AS
BEGIN
    SELECT  DISTINCT  Salary
    FROM EmployeeData;
END;


CREATE PROCEDURE EmployeeCount 
AS
BEGIN
    SELECT  COUNT(*)   as emp_count
    FROM EmployeeData;
END;

CREATE TABLE Projects(
    ProjectID INT  IDENTITY(1,1) Primary Key,
ProjectName VARCHAR(255),
Description  TEXT ,
StartDate DATE,
EndDate DATE ,
Status  VARCHAR(50) ,
OwnerUserID  INT ,
Budget DECIMAL(10, 2),
ClientID  INT ,
Location VARCHAR(255)
);

CREATE PROCEDURE GetAllProjects
 
AS
BEGIN
    SELECT *
    FROM Projects
  
END;
